#include <clQMC/clQMC.clh>
#define DIMENSION 30
clqmc_fptype simulateOneRun(StreamType* stream)
{
clqmc_fptype ret = 1.0;
for (uint j = 0; j < DIMENSION; j++) {
clqmc_fptype uj = nextCoordinate(stream);
ret *= 3 * uj * uj;
}
return ret;
}
